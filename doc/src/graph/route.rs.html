23fn new_router_id() -> usize {

27/// Manages routing queries for one profile. This structure uses contraction hierarchies to calculate
28/// routes very quickly. They are slower to construct, but fast to query.
29#[derive(Serialize, Deserialize)]
30pub struct Router {
31    #[serde(default = "new_router_id", skip_serializing)]
32    id: usize,
33    #[serde(deserialize_with = "deserialize_nodemap")]
34    node_map: NodeMap<IntersectionID>,
35    #[serde(serialize_with = "serialize_32", deserialize_with = "deserialize_32")]
36    ch: FastGraph,
37
38    pub closest_road: RTree<EdgeLocation>,
39}

41pub type EdgeLocation = GeomWithData<LineString, RoadID>;

52    /// Creates a router for a profile. This is slow to calculate, as it builds a
53    /// contraction hierarchy.
54    pub fn new(roads: &Vec<Road>, profile: ProfileID) -> Self {
55        let mut input_graph = InputGraph::new();
56        let mut node_map = NodeMap::new();
57        let mut duplicate_edges = HashSet::new();
58
59        for road in roads {
60            let cost = road.cost[profile.0].as_millis() as usize;
61            let node1 = node_map.get_or_insert(road.src_i);
62            let node2 = node_map.get_or_insert(road.dst_i);
63
64            // Loops aren't ever part of a shortest path. The fast_paths warning isn't actionable,
65            // so use our own
66            if node1 == node2 {
67                warn!("Loop on {} by {} won't be routeable", road.way, road.node1);
68                continue;
69            }
70
71            if road.allows_forwards(profile) {
72                input_graph.add_edge(node1, node2, cost);
73                if !duplicate_edges.insert((node1, node2)) {
74                    warn!(
75                        "Multiple edges from {} to {}; only one will be routeable",
76                        road.node1, road.node2
77                    );
78                }
79            }
80            if road.allows_backwards(profile) {
81                input_graph.add_edge(node2, node1, cost);
82                if !duplicate_edges.insert((node2, node1)) {
83                    warn!(
84                        "Multiple edges from {} to {}; only one will be routeable",
85                        road.node2, road.node1
86                    );
87                }
88            }
89        }
90        input_graph.freeze();
91        let ch = fast_paths::prepare(&input_graph);
92
93        let closest_road = RTree::bulk_load(
94            roads
95                .iter()
96                .filter(|r| r.access[profile.0] != Direction::None)
97                .map(|r| EdgeLocation::new(r.linestring.clone(), r.id))
98                .collect(),
99        );
100
101        Self {
102            id: new_router_id(),
103            node_map,
104            ch,
105            closest_road,
106        }
107    }

109    /// After the caller has manually updated per-road costs, this will recalculate the contraction
110    /// hierarchy. Note that access must remain the same!
111    pub fn update_costs(&mut self, roads: &Vec<Road>, profile: ProfileID) {
112        let mut input_graph = InputGraph::new();
113        let mut duplicate_edges = HashSet::new();
114        for road in roads {
115            let cost = road.cost[profile.0].as_millis() as usize;
116            let node1 = self
117                .node_map
118                .get(road.src_i)
119                .expect("new intersections somehow added");
120            let node2 = self
121                .node_map
122                .get(road.dst_i)
123                .expect("new intersections somehow added");
124
125            if node1 == node2 {
126                warn!("Loop on {} by {} won't be routeable", road.way, road.node1);
127                continue;
128            }
129
130            if road.allows_forwards(profile) {
131                input_graph.add_edge(node1, node2, cost);
132                if !duplicate_edges.insert((node1, node2)) {
133                    warn!(
134                        "Multiple edges from {} to {}; only one will be routeable",
135                        road.node1, road.node2
136                    );
137                }
138            }
139            if road.allows_backwards(profile) {
140                input_graph.add_edge(node2, node1, cost);
141                if !duplicate_edges.insert((node2, node1)) {
142                    warn!(
143                        "Multiple edges from {} to {}; only one will be routeable",
144                        road.node2, road.node1
145                    );
146                }
147            }
148        }
149        input_graph.freeze();
150
151        // The number of edges won't change, just costs
152        let node_ordering = self.ch.get_node_ordering();
153        let ch = fast_paths::prepare_with_order(&input_graph, &node_ordering)
154            .expect("prepare_with_order failed");
155        self.ch = ch;
156    }

158    /// Calculates a route between two positions.
159    pub fn route(&self, graph: &Graph, start: Position, end: Position) -> Result<Route> {
160        debug!("route from {start:?} to {end:?}");
161        if start == end {
162            bail!("start = end");
163        }
164
165        if start.road == end.road {
166            debug!("path: start = end road case");
167            return Ok(Route {
168                start,
169                end,
170                steps: vec![PathStep::Road {
171                    road: start.road,
172                    forwards: start.fraction_along < end.fraction_along,
173                }],
174            });
175        }
176
177        if start.intersection == end.intersection {
178            let common_intersection = start.intersection;
179            debug!("path: start = end intersection case");
180            let start_road = &graph.roads[start.road.0];
181            let end_road = &graph.roads[start.road.0];
182            return Ok(Route {
183                start,
184                end,
185                steps: vec![
186                    PathStep::Road {
187                        road: start.road,
188                        forwards: start_road.dst_i == common_intersection,
189                    },
190                    PathStep::Road {
191                        road: end.road,
192                        forwards: end_road.src_i == common_intersection,
193                    },
194                ],
195            });
196        }
197
198        let start_node = self.node_map.get(start.intersection).unwrap();
199        let end_node = self.node_map.get(end.intersection).unwrap();
200
201        let Some(path) = PATH_CALCULATORS.with(|calcs| {
202            calcs
203                .borrow_mut()
204                .entry(self.id)
205                .or_insert_with(|| fast_paths::create_calculator(&self.ch))
206                .calc_path(&self.ch, start_node, end_node)
207        }) else {
208            bail!("No path");
209        };
210
211        let mut steps = Vec::new();
212        for (pos, pair) in path.get_nodes().windows(2).with_position() {
213            let i1 = self.node_map.translate_id(pair[0]);
214            let i2 = self.node_map.translate_id(pair[1]);
215            let Some(road) = graph.find_edge(i1, i2) else {
216                bail!("No road between {i1:?} and {i2:?}");
217            };
218
219            if (pos == itertools::Position::First || pos == itertools::Position::Only)
220                && road.id != start.road
221            {
222                steps.push(PathStep::Road {
223                    road: start.road,
224                    // TODO Test carefully.
225                    forwards: start.fraction_along > 0.5,
226                });
227            }
228            steps.push(PathStep::Road {
229                road: road.id,
230                forwards: road.src_i == i1,
231            });
232            if (pos == itertools::Position::Last || pos == itertools::Position::Only)
233                && road.id != end.road
234            {
235                steps.push(PathStep::Road {
236                    road: end.road,
237                    // TODO Test carefully.
238                    forwards: end.fraction_along <= 0.5,
239                });
240            }
241        }
242
243        Ok(Route { start, end, steps })
244    }

246    /// Calculates a route between two intersections.
247    pub fn route_between_intersections(
248        &self,
249        graph: &Graph,
250        start_i: IntersectionID,
251        end_i: IntersectionID,
252    ) -> Result<Route> {
253        if start_i == end_i {
254            bail!("start = end");
255        }
256
257        let start_node = self.node_map.get(start_i).unwrap();
258        let end_node = self.node_map.get(end_i).unwrap();
259
260        let Some(path) = PATH_CALCULATORS.with(|calcs| {
261            calcs
262                .borrow_mut()
263                .entry(self.id)
264                .or_insert_with(|| fast_paths::create_calculator(&self.ch))
265                .calc_path(&self.ch, start_node, end_node)
266        }) else {
267            bail!("No path");
268        };
269
270        let mut steps = Vec::new();
271        for pair in path.get_nodes().windows(2) {
272            let i1 = self.node_map.translate_id(pair[0]);
273            let i2 = self.node_map.translate_id(pair[1]);
274            let Some(road) = graph.find_edge(i1, i2) else {
275                bail!("No road between {i1:?} and {i2:?}");
276            };
277
278            steps.push(PathStep::Road {
279                road: road.id,
280                forwards: road.src_i == i1,
281            });
282        }
283
284        let first_road = match steps[0] {
285            PathStep::Road { road, .. } => road,
286            _ => unreachable!(),
287        };
288        let start = Position {
289            intersection: start_i,
290            road: first_road,
291            fraction_along: if graph.roads[first_road.0].src_i == start_i {
292                0.0
293            } else {
294                1.0
295            },
296        };
297
298        let last_road = match steps.last().unwrap() {
299            PathStep::Road { road, .. } => *road,
300            _ => unreachable!(),
301        };
302        let end = Position {
303            intersection: end_i,
304            road: last_road,
305            fraction_along: if graph.roads[last_road.0].src_i == end_i {
306                0.0
307            } else {
308                1.0
309            },
310        };
311
312        Ok(Route { start, end, steps })
313    }

315    /// Calculate a route covering a sequence of waypoints. There may be spurs and doubling back.
316    pub fn route_between_many_intersections(
317        &self,
318        graph: &Graph,
319        waypoints: Vec<IntersectionID>,
320    ) -> Result<Route> {
321        if waypoints.len() < 2 {
322            bail!("Not enough waypoints");
323        }
324
325        let mut routes = Vec::new();
326        for pair in waypoints.windows(2) {
327            routes.push(self.route_between_intersections(graph, pair[0], pair[1])?);
328        }
329
330        let mut route = routes.remove(0);
331        for append in routes {
332            assert_eq!(route.end.intersection, append.start.intersection);
333            route.steps.extend(append.steps);
334            route.end = append.end;
335        }
336        Ok(route)
337    }

341    /// Renders a route as a linestring (in Mercator), with precise positions at the start and end.
342    pub fn linestring(&self, graph: &Graph) -> LineString {
343        self.split_linestrings(graph, |_| ()).pop().unwrap().0
344    }

346    /// Renders a route as a linestring (in Mercator), with precise positions at the start and end.
347    /// Optionally splits when some function on PathSteps produces a different value.
348    pub fn split_linestrings<T: Copy + PartialEq, F: Fn(RoadID) -> T>(
349        &self,
350        graph: &Graph,
351        key: F,
352    ) -> Vec<(LineString, T)> {
353        let mut results = Vec::new();
354
355        let mut pts = Vec::new();
356        let mut current_key = None;
357
358        for (pos, step) in self.steps.iter().with_position() {
359            match step {
360                PathStep::Road { road, forwards } => {
361                    let this_key = key(*road);
362                    if current_key.is_none() {
363                        current_key = Some(this_key);
364                    } else if current_key != Some(this_key) {
365                        // Something new
366                        pts.dedup();
367                        results.push((
368                            LineString::new(std::mem::take(&mut pts)),
369                            current_key.take().unwrap(),
370                        ));
371                        current_key = Some(this_key);
372                    }
373
374                    pts.extend(slice_road_step(
375                        &graph.roads[road.0].linestring,
376                        *forwards,
377                        &self.start,
378                        &self.end,
379                        pos,
380                    ));
381                }
382                PathStep::Transit { .. } => unreachable!(),
383            }
384        }
385
386        pts.dedup();
387        results.push((
388            LineString::new(std::mem::take(&mut pts)),
389            current_key.take().unwrap(),
390        ));
391        results
392    }
393
394    /// Returns the intersections in order. If the start or end steps are in the middle of the
395    /// road, snaps to the nearest intersection.
396    pub fn intersections(&self, graph: &Graph) -> Vec<IntersectionID> {
397        let mut result = vec![self.start.intersection];
398        for step in &self.steps {
399            if let PathStep::Road { road, forwards } = step {
400                let road = &graph.roads[road.0];
401                if *forwards {
402                    result.push(road.src_i);
403                    result.push(road.dst_i);
404                } else {
405                    result.push(road.dst_i);
406                    result.push(road.src_i);
407                }
408            }
409        }
410        result.push(self.end.intersection);
411        // The logic above is simpler by just adding everything, not worrying about pairs of
412        // things. Just dedupe here.
413        result.dedup();
414        result
415    }
416}

418fn slice_road_step(
419    linestring: &LineString,
420    forwards: bool,
421    start: &Position,
422    end: &Position,
423    pos: itertools::Position,
424) -> Vec<Coord> {
425    let mut pts = match pos {
426        itertools::Position::First => {
427            let (a, b) = if forwards {
428                (start.fraction_along, 1.0)
429            } else {
430                (0.0, start.fraction_along)
431            };
432            linestring
433                .line_split_twice(a, b)
434                .unwrap()
435                .into_second()
436                // TODO Workaround some crashes in Severance Snape, where the first PathStep seems
437                // to start on a road in reverse and immediately go somewhere else
438                .map(|ls| ls.0)
439                .unwrap_or_else(Vec::new)
440        }
441        itertools::Position::Last => {
442            let (a, b) = if forwards {
443                (0.0, end.fraction_along)
444            } else {
445                (end.fraction_along, 1.0)
446            };
447            linestring
448                .line_split_twice(a, b)
449                .unwrap()
450                .into_second()
451                .map(|ls| ls.0)
452                .unwrap_or_else(Vec::new)
453        }
454        itertools::Position::Middle => linestring.0.clone(),
455        itertools::Position::Only => linestring
456            .line_split_twice(start.fraction_along, end.fraction_along)
457            .unwrap()
458            .into_second()
459            .map(|ls| ls.0)
460            .unwrap_or_else(Vec::new),
461    };
462    if !forwards {
463        pts.reverse();
464    }
465    pts
