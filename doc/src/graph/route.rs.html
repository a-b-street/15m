13/// Manages routing queries for one profile. This structure uses contraction hierarchies to calculate
14/// routes very quickly. They are slower to construct, but fast to query.
15#[derive(Serialize, Deserialize)]
16pub struct Router {
17    #[serde(deserialize_with = "deserialize_nodemap")]
18    node_map: NodeMap<IntersectionID>,
19    #[serde(serialize_with = "serialize_32", deserialize_with = "deserialize_32")]
20    ch: FastGraph,
21    #[serde(skip_serializing, skip_deserializing)]
22    path_calc: RefCell<Option<PathCalculator>>,
23
24    pub closest_road: RTree<EdgeLocation>,
25}
26
27pub type EdgeLocation = GeomWithData<LineString, RoadID>;

29/// A route between two positions.

36impl Router {
37    /// Creates a router for a profile. This is slow to calculate, as it builds a
38    /// contraction hierarchy.
39    pub fn new(roads: &Vec<Road>, profile: ProfileID) -> Self {
40        let mut input_graph = InputGraph::new();
41        let mut node_map = NodeMap::new();
42
43        for road in roads {
44            let cost = road.cost[profile.0].as_millis() as usize;
45            let node1 = node_map.get_or_insert(road.src_i);
46            let node2 = node_map.get_or_insert(road.dst_i);
47
48            // Loops aren't ever part of a shortest path, and fast_paths warns loudly, so just skip
49            if node1 == node2 {
50                continue;
51            }
52
53            if road.allows_forwards(profile) {
54                input_graph.add_edge(node1, node2, cost);
55            }
56            if road.allows_backwards(profile) {
57                input_graph.add_edge(node2, node1, cost);
58            }
59        }
60        input_graph.freeze();
61        let ch = fast_paths::prepare(&input_graph);
62
63        let path_calc = RefCell::new(Some(fast_paths::create_calculator(&ch)));
64
65        let closest_road = RTree::bulk_load(
66            roads
67                .iter()
68                .filter(|r| r.access[profile.0] != Direction::None)
69                .map(|r| EdgeLocation::new(r.linestring.clone(), r.id))
70                .collect(),
71        );
72
73        Self {
74            node_map,
75            ch,
76            path_calc,
77
78            closest_road,
79        }
80    }

82    /// After the caller has manually updated per-road costs, this will recalculate the contraction
83    /// hierarchy. Note that access must remain the same!
84    pub fn update_costs(&mut self, roads: &Vec<Road>, profile: ProfileID) {
85        let mut input_graph = InputGraph::new();
86        for road in roads {
87            let cost = road.cost[profile.0].as_millis() as usize;
88            let node1 = self
89                .node_map
90                .get(road.src_i)
91                .expect("new intersections somehow added");
92            let node2 = self
93                .node_map
94                .get(road.dst_i)
95                .expect("new intersections somehow added");
96
97            // Loops aren't ever part of a shortest path, and fast_paths warns loudly, so just skip
98            if node1 == node2 {
99                continue;
100            }
101
102            if road.allows_forwards(profile) {
103                input_graph.add_edge(node1, node2, cost);
104            }
105            if road.allows_backwards(profile) {
106                input_graph.add_edge(node2, node1, cost);
107            }
108        }
109        input_graph.freeze();
110
111        let node_ordering = self.ch.get_node_ordering();
112        let ch = fast_paths::prepare_with_order(&input_graph, &node_ordering)
113            .expect("prepare_with_order failed");
114        self.ch = ch;
115    }

117    /// Calculates a route between two positions.
118    pub fn route(&self, graph: &Graph, start: Position, end: Position) -> Result<Route> {
119        debug!("route from {start:?} to {end:?}");
120        if start == end {
121            bail!("start = end");
122        }
123
124        if start.road == end.road {
125            debug!("path: start = end road case");
126            return Ok(Route {
127                start,
128                end,
129                steps: vec![PathStep::Road {
130                    road: start.road,
131                    forwards: start.fraction_along < end.fraction_along,
132                }],
133            });
134        }
135
136        if start.intersection == end.intersection {
137            let common_intersection = start.intersection;
138            debug!("path: start = end intersection case");
139            let start_road = &graph.roads[start.road.0];
140            let end_road = &graph.roads[start.road.0];
141            return Ok(Route {
142                start,
143                end,
144                steps: vec![
145                    PathStep::Road {
146                        road: start.road,
147                        forwards: start_road.dst_i == common_intersection,
148                    },
149                    PathStep::Road {
150                        road: end.road,
151                        forwards: end_road.src_i == common_intersection,
152                    },
153                ],
154            });
155        }
156
157        let start_node = self.node_map.get(start.intersection).unwrap();
158        let end_node = self.node_map.get(end.intersection).unwrap();
159
160        let Some(path) = self
161            .path_calc
162            .borrow_mut()
163            // This'll be empty right after loading a serialized Graph
164            .get_or_insert_with(|| fast_paths::create_calculator(&self.ch))
165            .calc_path(&self.ch, start_node, end_node)
166        else {
167            bail!("No path");
168        };
169
170        let mut steps = Vec::new();
171        for (pos, pair) in path.get_nodes().windows(2).with_position() {
172            let i1 = self.node_map.translate_id(pair[0]);
173            let i2 = self.node_map.translate_id(pair[1]);
174            let Some(road) = graph.find_edge(i1, i2) else {
175                bail!("No road between {i1:?} and {i2:?}");
176            };
177
178            if (pos == itertools::Position::First || pos == itertools::Position::Only)
179                && road.id != start.road
180            {
181                steps.push(PathStep::Road {
182                    road: start.road,
183                    // TODO Test carefully.
184                    forwards: start.fraction_along > 0.5,
185                });
186            }
187            steps.push(PathStep::Road {
188                road: road.id,
189                forwards: road.src_i == i1,
190            });
191            if (pos == itertools::Position::Last || pos == itertools::Position::Only)
192                && road.id != end.road
193            {
194                steps.push(PathStep::Road {
195                    road: end.road,
196                    // TODO Test carefully.
197                    forwards: end.fraction_along <= 0.5,
198                });
199            }
200        }
201
202        Ok(Route { start, end, steps })
203    }

205    /// Calculates a route between two intersections.
206    pub fn route_between_intersections(
207        &self,
208        graph: &Graph,
209        start_i: IntersectionID,
210        end_i: IntersectionID,
211    ) -> Result<Route> {
212        if start_i == end_i {
213            bail!("start = end");
214        }
215
216        let start_node = self.node_map.get(start_i).unwrap();
217        let end_node = self.node_map.get(end_i).unwrap();
218
219        let Some(path) = self
220            .path_calc
221            .borrow_mut()
222            // This'll be empty right after loading a serialized Graph
223            .get_or_insert_with(|| fast_paths::create_calculator(&self.ch))
224            .calc_path(&self.ch, start_node, end_node)
225        else {
226            bail!("No path");
227        };
228
229        let mut steps = Vec::new();
230        for pair in path.get_nodes().windows(2) {
231            let i1 = self.node_map.translate_id(pair[0]);
232            let i2 = self.node_map.translate_id(pair[1]);
233            let Some(road) = graph.find_edge(i1, i2) else {
234                bail!("No road between {i1:?} and {i2:?}");
235            };
236
237            steps.push(PathStep::Road {
238                road: road.id,
239                forwards: road.src_i == i1,
240            });
241        }
242
243        let first_road = match steps[0] {
244            PathStep::Road { road, .. } => road,
245            _ => unreachable!(),
246        };
247        let start = Position {
248            intersection: start_i,
249            road: first_road,
250            fraction_along: if graph.roads[first_road.0].src_i == start_i {
251                0.0
252            } else {
253                1.0
254            },
255        };
256
257        let last_road = match steps.last().unwrap() {
258            PathStep::Road { road, .. } => *road,
259            _ => unreachable!(),
260        };
261        let end = Position {
262            intersection: end_i,
263            road: last_road,
264            fraction_along: if graph.roads[last_road.0].src_i == end_i {
265                0.0
266            } else {
267                1.0
268            },
269        };
270
271        Ok(Route { start, end, steps })
272    }

274    /// Calculate a route covering a sequence of waypoints. There may be spurs and doubling back.
275    pub fn route_between_many_intersections(
276        &self,
277        graph: &Graph,
278        waypoints: Vec<IntersectionID>,
279    ) -> Result<Route> {
280        if waypoints.len() < 2 {
281            bail!("Not enough waypoints");
282        }
283
284        let mut routes = Vec::new();
285        for pair in waypoints.windows(2) {
286            routes.push(self.route_between_intersections(graph, pair[0], pair[1])?);
287        }
288
289        let mut route = routes.remove(0);
290        for append in routes {
291            assert_eq!(route.end.intersection, append.start.intersection);
292            route.steps.extend(append.steps);
293            route.end = append.end;
294        }
295        Ok(route)
296    }

300    /// Renders a route as a linestring (in Mercator), with precise positions at the start and end.
301    pub fn linestring(&self, graph: &Graph) -> LineString {
302        self.split_linestrings(graph, |_| ()).pop().unwrap().0
303    }

305    /// Renders a route as a linestring (in Mercator), with precise positions at the start and end.
306    /// Optionally splits when some function on PathSteps produces a different value.
307    pub fn split_linestrings<T: Copy + PartialEq, F: Fn(RoadID) -> T>(
308        &self,
309        graph: &Graph,
310        key: F,
311    ) -> Vec<(LineString, T)> {
312        let mut results = Vec::new();
313
314        let mut pts = Vec::new();
315        let mut current_key = None;
316
317        for (pos, step) in self.steps.iter().with_position() {
318            match step {
319                PathStep::Road { road, forwards } => {
320                    let this_key = key(*road);
321                    if current_key.is_none() {
322                        current_key = Some(this_key);
323                    } else if current_key != Some(this_key) {
324                        // Something new
325                        pts.dedup();
326                        results.push((
327                            LineString::new(std::mem::take(&mut pts)),
328                            current_key.take().unwrap(),
329                        ));
330                        current_key = Some(this_key);
331                    }
332
333                    pts.extend(slice_road_step(
334                        &graph.roads[road.0].linestring,
335                        *forwards,
336                        &self.start,
337                        &self.end,
338                        pos,
339                    ));
340                }
341                PathStep::Transit { .. } => unreachable!(),
342            }
343        }
344
345        pts.dedup();
346        results.push((
347            LineString::new(std::mem::take(&mut pts)),
348            current_key.take().unwrap(),
349        ));
350        results
351    }
352
353    /// Returns the intersections in order. If the start or end steps are in the middle of the
354    /// road, snaps to the nearest intersection.
355    pub fn intersections(&self, graph: &Graph) -> Vec<IntersectionID> {
356        let mut result = vec![self.start.intersection];
357        for step in &self.steps {
358            if let PathStep::Road { road, forwards } = step {
359                let road = &graph.roads[road.0];
360                if *forwards {
361                    result.push(road.src_i);
362                    result.push(road.dst_i);
363                } else {
364                    result.push(road.dst_i);
365                    result.push(road.src_i);
366                }
367            }
368        }
369        result.push(self.end.intersection);
370        // The logic above is simpler by just adding everything, not worrying about pairs of
371        // things. Just dedupe here.
372        result.dedup();
373        result
374    }
375}

377fn slice_road_step(
378    linestring: &LineString,
379    forwards: bool,
380    start: &Position,
381    end: &Position,
382    pos: itertools::Position,
383) -> Vec<Coord> {
384    let mut pts = match pos {
385        itertools::Position::First => {
386            let (a, b) = if forwards {
387                (start.fraction_along, 1.0)
388            } else {
389                (0.0, start.fraction_along)
390            };
391            linestring
392                .line_split_twice(a, b)
393                .unwrap()
394                .into_second()
395                // TODO Workaround some crashes in Severance Snape, where the first PathStep seems
396                // to start on a road in reverse and immediately go somewhere else
397                .map(|ls| ls.0)
398                .unwrap_or_else(Vec::new)
399        }
400        itertools::Position::Last => {
401            let (a, b) = if forwards {
402                (0.0, end.fraction_along)
403            } else {
404                (end.fraction_along, 1.0)
405            };
406            linestring
407                .line_split_twice(a, b)
408                .unwrap()
409                .into_second()
410                .map(|ls| ls.0)
411                .unwrap_or_else(Vec::new)
412        }
413        itertools::Position::Middle => linestring.0.clone(),
414        itertools::Position::Only => linestring
415            .line_split_twice(start.fraction_along, end.fraction_along)
416            .unwrap()
417            .into_second()
418            .map(|ls| ls.0)
419            .unwrap_or_else(Vec::new),
420    };
421    if !forwards {
422        pts.reverse();
423    }
424    pts
