14    pub fn transit_route_gj(
15        &self,
16        start: Position,
17        end: Position,
18        // TODO Parameterizing this function gets messy, but splitting into two separate doesn't seem
19        // like a good idea yet
20        debug_search: bool,
21        use_heuristic: bool,
22        start_time: NaiveTime,
23        mut timer: Timer,
24    ) -> Result<String> {
25        if start == end {
26            bail!("start = end");
27        }
28        let Some(profile) = self.walking_profile_for_transit else {
29            bail!("public transit hasn't been set up");
30        };
31        // TODO Handle start.road == end.road case. Share code somewhere.
32
33        let end_pt = self.intersections[end.intersection.0].point;
34        // TODO Share constant properly
35        // TODO Think through if this is admissible and/or consistent
36        let heuristic = |i: IntersectionID| {
37            if use_heuristic {
38                Duration::from_secs_f64(
39                    Euclidean.distance(self.intersections[i.0].point, end_pt) / 1.34112,
40                )
41            } else {
42                Duration::ZERO
43            }
44        };
45
46        // TODO stops are associated with roads, so the steps using transit are going to look a little
47        // weird / be a little ambiguous
48        //
49        // or rethink the nodes and edges in the graph. nodes are pathsteps -- a road in some direction
50        // or a transit thing. an edge is a turn or a transition to/from transit
51
52        let mut backrefs: HashMap<IntersectionID, Backreference> = HashMap::new();
53
54        // If we're debugging the search process, just remember the order of visited nodes
55        let mut search_record: Vec<IntersectionID> = Vec::new();
56
57        timer.step("dijkstra");
58        // Store the actual cost/time to reach somewhere as the item. Include a heuristic
59        let mut queue: BinaryHeap<PriorityQueueItem<NaiveTime, (IntersectionID, NaiveTime)>> =
60            BinaryHeap::new();
61        queue.push(PriorityQueueItem::new(
62            start_time + heuristic(start.intersection),
63            (start.intersection, start_time),
64        ));
65
66        while let Some(current) = queue.pop() {
67            // Don't use current.cost, since it might include a heuristic
68            let (current_i, current_time) = current.value;
69            if current_i == end.intersection {
70                if debug_search {
71                    return render_debug(search_record, backrefs, self, timer);
72                } else {
73                    return render_path(backrefs, self, start, end, timer);
74                }
75            }
76            if debug_search {
77                search_record.push(current_i);
78            }
79
80            for r in &self.intersections[current_i.0].roads {
81                let road = &self.roads[r.0];
82
83                // Handle walking to the other end of the road
84                let total_cost = current_time + road.cost[profile.0];
85                if road.src_i == current_i && road.allows_forwards(profile) {
86                    if let Entry::Vacant(entry) = backrefs.entry(road.dst_i) {
87                        entry.insert(Backreference {
88                            src_i: current_i,
89                            step: PathStep::Road {
90                                road: *r,
91                                forwards: true,
92                            },
93                            time1: current_time,
94                            time2: total_cost,
95                        });
96                        queue.push(PriorityQueueItem::new(
97                            total_cost + heuristic(road.dst_i),
98                            (road.dst_i, total_cost),
99                        ));
100                    }
101                } else if road.dst_i == current_i && road.allows_backwards(profile) {
102                    if let Entry::Vacant(entry) = backrefs.entry(road.src_i) {
103                        entry.insert(Backreference {
104                            src_i: current_i,
105                            step: PathStep::Road {
106                                road: *r,
107                                forwards: false,
108                            },
109                            time1: current_time,
110                            time2: total_cost,
111                        });
112                        queue.push(PriorityQueueItem::new(
113                            total_cost + heuristic(road.src_i),
114                            (road.src_i, total_cost),
115                        ));
116                    }
117                }
118
119                // Use transit!
120                for stop1 in &road.stops {
121                    // Find all trips leaving from this step in the next 30 minutes
122                    // TODO Figure out how to prune that search time better
123                    for next_step in
124                        self.gtfs
125                            .trips_from(*stop1, current_time, Duration::from_secs(30 * 60))
126                    {
127                        // TODO Here's the awkwardness -- arrive at both the intersections for that
128                        // road
129                        let stop2_road = &self.roads[self.gtfs.stops[next_step.stop2.0].road.0];
130                        for i in [stop2_road.src_i, stop2_road.dst_i] {
131                            if let Entry::Vacant(entry) = backrefs.entry(i) {
132                                entry.insert(Backreference {
133                                    src_i: current_i,
134                                    step: PathStep::Transit {
135                                        stop1: *stop1,
136                                        trip: next_step.trip,
137                                        stop2: next_step.stop2,
138                                    },
139                                    time1: next_step.time1,
140                                    time2: next_step.time2,
141                                });
142                                queue.push(PriorityQueueItem::new(
143                                    next_step.time2 + heuristic(i),
144                                    (i, next_step.time2),
145                                ));
146                            }
147                        }
148                    }
149                }
150            }
151        }
152
153        bail!("No path found");
154    }

157// How'd we get somewhere?
158struct Backreference {
159    // Where were we at the beginning of this step?
160    src_i: IntersectionID,
161    step: PathStep,
162    // When'd we start this step?
163    time1: NaiveTime,
164    // When'd we finish?
165    time2: NaiveTime,
166}

168fn render_path(
169    mut backrefs: HashMap<IntersectionID, Backreference>,
170    graph: &Graph,
171    start: Position,
172    end: Position,
173    mut timer: Timer,
174) -> Result<String> {
175    timer.step("render");
176
177    // Just get PathSteps in order first (Step, time1, time2)
178    let mut steps: Vec<(PathStep, NaiveTime, NaiveTime)> = Vec::new();
179    let mut at = end.intersection;
180    loop {
181        if at == start.intersection {
182            break;
183        }
184        let backref = backrefs.remove(&at).unwrap();
185        steps.push((backref.step, backref.time1, backref.time2));
186        at = backref.src_i;
187    }
188    steps.reverse();
189
190    // Assemble PathSteps into features. Group road and transit steps together
191    let mut features = Vec::new();
192    for chunk in steps.chunk_by(|a, b| match (&a.0, &b.0) {
193        (PathStep::Road { .. }, PathStep::Road { .. }) => true,
194        (PathStep::Transit { trip: trip1, .. }, PathStep::Transit { trip: trip2, .. }) => {
195            trip1 == trip2
196        }
197        _ => false,
198    }) {
199        let mut pts = Vec::new();
200        let mut num_stops = 0;
201        let mut trip_id = None;
202        for (step, _, _) in chunk {
203            match step {
204                PathStep::Road { road, forwards } => {
205                    let road = &graph.roads[road.0];
206                    if *forwards {
207                        pts.extend(road.linestring.0.clone());
208                    } else {
209                        let mut rev = road.linestring.0.clone();
210                        rev.reverse();
211                        pts.extend(rev);
212                    }
213                }
214                PathStep::Transit { stop1, stop2, trip } => {
215                    trip_id = Some(trip);
216                    num_stops += 1;
217                    pts.push(graph.gtfs.stops[stop1.0].point.into());
218                    pts.push(graph.gtfs.stops[stop2.0].point.into());
219                }
220            }
221        }
222        pts.dedup();
223
224        let mut f = Feature::from(Geometry::from(
225            &graph.mercator.to_wgs84(&LineString::new(pts)),
226        ));
227        f.set_property("time1", chunk[0].1.to_string());
228        f.set_property("time2", chunk.last().unwrap().2.to_string());
229
230        if let Some(trip) = trip_id {
231            f.set_property("kind", "transit");
232            f.set_property("trip", trip.0);
233            f.set_property(
234                "route",
235                graph.gtfs.routes[graph.gtfs.trips[trip.0].route.0].describe(),
236            );
237            f.set_property("num_stops", num_stops);
238        } else {
239            f.set_property("kind", "road");
240        }
241        features.push(f);
242    }
243    timer.done();
244    Ok(serde_json::to_string(&GeoJson::from(features))?)
245}

247fn render_debug(
248    search_record: Vec<IntersectionID>,
249    mut backrefs: HashMap<IntersectionID, Backreference>,
250    graph: &Graph,
251    mut timer: Timer,
252) -> Result<String> {
253    timer.step("render");
254    // Create a FeatureCollection with the nodes searched, in order. Pairs of linestrings (a step
255    // to get somewhere) and points (for the intersection)
256    let mut features = Vec::new();
257    // Skip the first node, because it'll have no backreference
258    for i in search_record.into_iter().skip(1) {
259        let backref = backrefs.remove(&i).unwrap();
260        match backref.step {
261            PathStep::Road { road, .. } => {
262                let mut f = Feature::from(Geometry::from(
263                    &graph.mercator.to_wgs84(&graph.roads[road.0].linestring),
264                ));
265                f.set_property("kind", "road");
266                features.push(f);
267            }
268            PathStep::Transit { stop1, stop2, .. } => {
269                let mut f = Feature::from(Geometry::from(&graph.mercator.to_wgs84(
270                    &LineString::new(vec![
271                        graph.gtfs.stops[stop1.0].point.into(),
272                        graph.gtfs.stops[stop2.0].point.into(),
273                    ]),
274                )));
275                f.set_property("kind", "transit");
276                features.push(f);
277            }
278        }
279
280        let mut f = Feature::from(Geometry::from(
281            &graph.mercator.to_wgs84(&graph.intersections[i.0].point),
282        ));
283        f.set_property("time", backref.time2.to_string());
284        features.push(f);
285    }
286
287    let json = serde_json::to_string(&GeoJson::from(features))?;
288    timer.done();
289    Ok(json)
