19    fn snap_by_endpoints(&self, input: &LineString, profile: ProfileID) -> Result<Route> {
20        // Simple start: just match the endpoints and find the optimal route, according to
21        // that profile's graph.
22        let start = self.snap_to_road(*input.coords().next().unwrap(), profile);
23        let end = self.snap_to_road(*input.coords().last().unwrap(), profile);
24        let route = self.routers[profile.0].route(self, start, end)?;

26        // TODO Detect/handle zero-length output here

31    fn snap_greedy(&self, input: &LineString, profile: ProfileID) -> Result<Route> {
32        let start = self.snap_to_road(*input.coords().next().unwrap(), profile);
33        let end = self.snap_to_road(*input.coords().last().unwrap(), profile);
34
35        let mut current = start.intersection;
36        let mut fraction_along = 0.0;
37        let mut steps = Vec::new();
38
39        while current != end.intersection {
40            match self
41                .next_steps(current)
42                .filter_map(|(i, step)| {
43                    // Find the closest point on the input linestring to this possible next
44                    // intersection
45                    match input.closest_point(&self.intersections[i.0].point) {
46                        Closest::Intersection(pt) | Closest::SinglePoint(pt) => {
47                            // How far along on the input linestring is it? If we'd move backwards,
48                            // skip it
49                            let new_fraction_along = input.line_locate_point(&pt)?;
50                            if new_fraction_along > fraction_along {
51                                let dist = (100.0
52                                    * Euclidean.distance(pt, self.intersections[i.0].point))
53                                    as usize;
54                                Some((i, step, dist, new_fraction_along))
55                            } else {
56                                None
57                            }
58                        }
59                        Closest::Indeterminate => None,
60                    }
61                })
62                // TODO Maybe also use new_fraction_along to judge the best next step
63                .min_by_key(|(_, _, dist, _)| *dist)
64            {
65                Some((i, step, _, new_fraction_along)) => {
66                    fraction_along = new_fraction_along;
67                    steps.push(step);
68                    current = i;
69                }
70                None => bail!("Got stuck at {}", self.intersections[current.0].node),
71            }
72        }
73
74        Ok(Route { steps, start, end })
75    }
76
77    // TODO Ignores profile and direction
78    fn next_steps(
79        &self,
80        i: IntersectionID,
81    ) -> impl Iterator<Item = (IntersectionID, PathStep)> + '_ {
82        self.intersections[i.0].roads.iter().flat_map(move |r| {
83            let road = &self.roads[r.0];
84            if road.src_i != i {
85                Some((
86                    road.src_i,
87                    PathStep::Road {
88                        road: *r,
89                        forwards: false,
90                    },
91                ))
92            } else if road.dst_i != i {
93                Some((
94                    road.dst_i,
95                    PathStep::Road {
96                        road: *r,
97                        forwards: true,
98                    },
99                ))
100            } else {
101                // A loop on i
102                None
103            }
104        })
105    }

109pub fn score_similarity(ls1: &LineString, ls2: &LineString) -> Option<(f64, f64)> {
110    // Just check length
111    let len1 = Euclidean.length(ls1);
112    let len2 = Euclidean.length(ls2);
113    let len_pct = if len1 < len2 {
114        len2 / len1
115    } else {
116        len1 / len2
117    };
118
119    // Walk along each and take the distance between (hopefully equivalent) points. This only
120    // really makes sense if the lengths are the same. Not sure how to scale this.
121    let mut dist_between_equiv_pts = 0.0;
122    for pct in 0..=100 {
123        let pct = (pct as f64) / 100.0;
124        // TODO Where do we have zero-length lines?
125        let pt1 = ls1.point_at_ratio_from_start(&Euclidean, pct)?;
126        let pt2 = ls2.point_at_ratio_from_start(&Euclidean, pct)?;
127
128        dist_between_equiv_pts += Euclidean.distance(pt1, pt2);
129    }
130
131    Some((len_pct, dist_between_equiv_pts))
