1use chrono::NaiveTime;
2use std::collections::{BinaryHeap, HashMap, HashSet};
3use std::time::Duration;
4
5use utils::PriorityQueueItem;
6
7use crate::{Graph, IntersectionID, ProfileID, RoadID};
8
9impl Graph {
10    // TODO Doesn't account for start/end distance along roads
11    /// From a list of start intersections, floods out the graph for a profile until `end_time` is
12    /// reached. Returns the time needed to reach each road within that range. This query is not
13    /// precise about positions along a road.
14    pub fn get_costs(
15        &self,
16        starts: Vec<IntersectionID>,
17        profile: ProfileID,
18        public_transit: bool,
19        start_time: NaiveTime,
20        end_time: NaiveTime,
21    ) -> HashMap<RoadID, Duration> {
22        let mut visited: HashSet<IntersectionID> = HashSet::new();
23        let mut cost_per_road: HashMap<RoadID, Duration> = HashMap::new();
24        let mut queue: BinaryHeap<PriorityQueueItem<NaiveTime, IntersectionID>> = BinaryHeap::new();
25
26        for start in starts {
27            queue.push(PriorityQueueItem::new(start_time, start));
28        }
29
30        while let Some(current) = queue.pop() {
31            if visited.contains(&current.value) {
32                continue;
33            }
34            visited.insert(current.value);
35            if current.cost > end_time {
36                continue;
37            }
38
39            for r in &self.intersections[current.value.0].roads {
40                let road = &self.roads[r.0];
41                let total_cost = current.cost + road.cost[profile.0];
42                cost_per_road
43                    .entry(*r)
44                    .or_insert((total_cost - start_time).to_std().unwrap());
45
46                if road.src_i == current.value && road.allows_forwards(profile) {
47                    queue.push(PriorityQueueItem::new(total_cost, road.dst_i));
48                }
49                if road.dst_i == current.value && road.allows_backwards(profile) {
50                    queue.push(PriorityQueueItem::new(total_cost, road.src_i));
51                }
52
53                if public_transit {
54                    for stop1 in &road.stops {
55                        // Find all trips leaving from this step before the end_time
56                        for next_step in self.gtfs.trips_from(
57                            *stop1,
58                            current.cost,
59                            (end_time - current.cost).to_std().unwrap(),
60                        ) {
61                            // TODO Awkwardly, arrive at both intersections for the next stop's road
62                            let stop2_road = &self.roads[self.gtfs.stops[next_step.stop2.0].road.0];
63                            for i in [stop2_road.src_i, stop2_road.dst_i] {
64                                queue.push(PriorityQueueItem::new(next_step.time2, i));
65                            }
66                        }
67                    }
68                }
69            }
70        }
71
72        cost_per_road
73    }
74}
