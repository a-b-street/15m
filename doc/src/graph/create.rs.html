12impl Graph {
13    /// Constructs a graph from OpenStreetMap data.
14    ///
15    /// - `input_bytes`: Bytes of an osm.pbf or osm.xml file
16    /// - `osm_reader`: A callback for every OSM element read, to extract non-graph data
17    /// - `post_process_graph`: A callback to remove edges and intersections after initially
18    ///   importing.
19    /// - `scrape_graph`: A callback to capture anything from osm2graph that's otherwise lost. It
20    ///   can be stored on the `osm_reader` struct.
21    /// - `profiles`: A list of named profiles. Each one assigns an access direction and cost,
22    ///   given OSM tags and a Euclidean center-line. If every profile assigns `Direction::None`,
23    ///   then the Road is completely excluded from the graph.
24    pub fn new<R: utils::osm2graph::OsmReader>(
25        input_bytes: &[u8],
26        osm_reader: &mut R,
27        post_process_graph: Box<dyn Fn(&mut utils::osm2graph::Graph) -> Result<()>>,
28        scrape_graph: Box<dyn Fn(&mut R, &utils::osm2graph::Graph) -> Result<()>>,
29        profiles: Vec<(
30            String,
31            Box<dyn Fn(&Tags, &LineString) -> (Direction, Duration)>,
32        )>,
33        timer: &mut Timer,
34    ) -> Result<Graph> {
35        timer.step("parse OSM and split graph");
36
37        let mut graph = utils::osm2graph::Graph::new(
38            input_bytes,
39            |tags| {
40                if !tags.has("highway") || tags.is("highway", "proposed") || tags.is("area", "yes")
41                {
42                    return false;
43                }
44                // Make sure at least one profile allows access
45                // TODO It's weird to pass in an empty linestring
46                // TODO It's inefficient to call the profiles twice
47                let empty = LineString::new(Vec::new());
48                profiles
49                    .iter()
50                    .any(|(_, profile)| profile(tags, &empty).0 != Direction::None)
51            },
52            osm_reader,
53        )?;
54        post_process_graph(&mut graph)?;
55        graph.compact_ids();
56        scrape_graph(osm_reader, &graph)?;
57
58        timer.step("calculate road attributes");
59        let mut roads: Vec<Road> = graph
60            .edges
61            .into_values()
62            .map(|e| Road {
63                id: RoadID(e.id.0),
64                src_i: IntersectionID(e.src.0),
65                dst_i: IntersectionID(e.dst.0),
66                way: e.osm_way,
67                node1: e.osm_node1,
68                node2: e.osm_node2,
69                osm_tags: e.osm_tags,
70                length_meters: Euclidean.length(&e.linestring),
71                linestring: e.linestring,
72
73                access: Vec::new(),
74                cost: Vec::new(),
75                stops: Vec::new(),
76            })
77            .collect();
78
79        // Copy all the fields
80        let intersections: Vec<Intersection> = graph
81            .intersections
82            .into_values()
83            .map(|mut i| {
84                // Sort intersection roads clockwise, starting from North
85                i.edges.sort_by_cached_key(|edge_id| {
86                    let road = &roads[edge_id.0];
87                    let bearing = bearing_from_endpoint(i.point, &road.linestring);
88                    // work around that f64 is not Ord
89                    debug_assert!(
90                        bearing.is_finite(),
91                        "Assuming bearing output is always 0...360, this shouldn't happen"
92                    );
93                    (bearing * 1e6) as i64
94                });
95
96                Intersection {
97                    id: IntersectionID(i.id.0),
98                    point: i.point,
99                    node: i.osm_node,
100                    roads: i.edges.into_iter().map(|e| RoadID(e.0)).collect(),
101                }
102            })
103            .collect();
104
105        timer.step("set up profiles");
106        for road in &mut roads {
107            let mut access = Vec::new();
108            let mut cost = Vec::new();
109            for (_, profile) in &profiles {
110                let (dir, c) = profile(&road.osm_tags, &road.linestring);
111                access.push(dir);
112                cost.push(c);
113            }
114            road.access = access;
115            road.cost = cost;
116        }
117
118        timer.push("building routers");
119        let mut routers = Vec::new();
120        let mut profile_names = BTreeMap::new();
121        for (idx, (name, _)) in profiles.into_iter().enumerate() {
122            timer.step(format!("for {name}"));
123            routers.push(Router::new(&roads, ProfileID(idx)));
124
125            profile_names.insert(name, ProfileID(idx));
126        }
127        timer.pop();
128
129        Ok(Graph {
130            roads,
131            intersections,
132            mercator: graph.mercator,
133            profile_names,
134            walking_profile_for_transit: None,
135            routers,
136            boundary_polygon: graph.boundary_polygon,
137
138            gtfs: GtfsModel::empty(),
139        })
140    }

142    /// Adds in GTFS data to the current graph. This only makes sense to call once.
143    #[cfg(feature = "gtfs")]
144    pub async fn setup_gtfs(
145        &mut self,
146        source: crate::GtfsSource,
147        profile: ProfileID,
148        timer: &mut Timer,
149    ) -> Result<()> {
150        if self.walking_profile_for_transit.is_some() {
151            bail!("Can't call setup_gtfs twice");
152        }
153        self.walking_profile_for_transit = Some(profile);
154
155        use crate::GtfsSource;
156
157        timer.push("setting up GTFS");
158        timer.step("parse");
159        let mut gtfs = match source {
160            GtfsSource::Dir(path) => GtfsModel::parse(&path, Some(&self.mercator))?,
161            GtfsSource::Geomedea(url) => GtfsModel::from_geomedea(&url, &self.mercator).await?,
162            GtfsSource::None => GtfsModel::empty(),
163        };
164        snap_stops(&mut self.roads, &mut gtfs, &self.routers[profile.0], timer);
165        self.gtfs = gtfs;
166        timer.pop();
167        Ok(())
168    }
169}

172fn snap_stops(
173    roads: &mut Vec<Road>,
174    gtfs: &mut GtfsModel,
175    foot_router: &Router,
176    timer: &mut Timer,
177) {
178    if gtfs.stops.is_empty() {
179        return;
180    }
181
182    timer.step(format!(
183        "find closest roads per stop ({})",
184        gtfs.stops.len()
185    ));
186    // TODO Make an iterator method that returns the IDs too
187    for (idx, stop) in gtfs.stops.iter_mut().enumerate() {
188        let stop_id = crate::gtfs::StopID(idx);
189        if let Some(r) = foot_router
190            .closest_road
191            .nearest_neighbor(&stop.point.into())
192        {
193            // TODO Limit how far away we snap, or use the boundary polygon
194            roads[r.data.0].stops.push(stop_id);
195            stop.road = r.data;
196        } else {
197            // TODO Need to get rid of the stop
198            error!("{stop_id:?} didn't snap to any road");
199        }
200    }
201}

206/// The bearing of the first segment of `linestring` starting from `endpoint`.
207///
208/// precondition: `endpoint` must be either the first or last point in `linestring`
209/// precondition: `linestring` must have at least 2 coordinates
210fn bearing_from_endpoint(endpoint: Point, linestring: &LineString) -> f64 {
211    assert!(
212        linestring.0.len() >= 2,
213        "zero length roads should be filtered out"
214    );
215    let next_coord = if endpoint.0 == linestring.0[0] {
216        linestring.0[1]
217    } else if endpoint.0 == linestring.0[linestring.0.len() - 1] {
218        linestring.0[linestring.0.len() - 2]
219    } else {
220        // I'm assuming this won't happen, but maybe it's possible,
221        // e.g. to different rounding schemes.
222        debug_assert!(false, "road does not terminate at intersection");
223        linestring.0[1]
224    };
225
226    euclidean_bearing(endpoint.0, next_coord)
227}
228
229fn euclidean_bearing(origin: Coord, destination: Coord) -> f64 {
230    (angle_of_line(Line::new(origin, destination)) + 450.0) % 360.0
231}
232
233fn angle_of_line(line: Line) -> f64 {
234    (line.dy()).atan2(line.dx()).to_degrees()
